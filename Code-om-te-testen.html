Trainset en Test aanmaken

create_train_test <- function(student_math, size = 0.8, train = TRUE) {
    n_row = nrow(student_math)
    total_row = size * n_row
    train_sample <- 1: total_row
    if (train == TRUE) {
        return (student_math[train_sample, ])
    } else {
        return (student_math[-train_sample, ])
    }
}
